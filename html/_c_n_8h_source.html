105 std::vector<double> defineRate(double dr_ = 0) {
106 std::vector<double> result(Nt, 0.0);
107 double dt = (Tf - T0) / Nt;
108
109 for (int i = 0; i < Nt; i++) {
110 double t = T0 + i * dt;
111 auto it = std::find_if(r.begin(), r.end(),
112 [t](const std::pair<double, double>& pair) { return std::abs(pair.first - t) < 1e-9; });
113
114 if (it != r.end()) {
115 result[i] = it->second + dr_;
116 continue;
117 }
118 auto upper = std::upper_bound(r.begin(), r.end(), t,
119 [](double val, const auto& pair) { return val < pair.first; });
120
121 if (upper == r.end()) {
122 result[i] = std::prev(r.end())->second + dr_;;
123 continue;
124 }
125 if (upper == r.begin()) {
126 result[i] = r.begin()->second + dr_;
127 continue;
128 }
129
130 auto lower = std::prev(upper);
131 double x1 = lower->first, v1 = lower->second;
132 double x2 = upper->first, v2 = upper->second;
133 result[i] = dr_ + v1 + (v2 - v1) * (t - x1) / (x2 - x1);
134
135 }
136 return result;
137 }
145 void solveTridiagonal(const std::vector<double>& lower, const std::vector<double>& diag,
146 const std::vector<double>& upper, const std::vector<double>& b,
147 std::vector<double>& result) {
148 int n = diag.size();
149 std::vector<double> c_prime(n, 0.0);
150 std::vector<double> d_prime(n, 0.0);
151 result.resize(n);
152
153 //forward step
154 c_prime[0] = upper[0] / diag[0];
155 d_prime[0] = b[0] / diag[0];
156 for (int i = 1; i < n; ++i) {
157 double m = diag[i] - lower[i - 1] * c_prime[i - 1];
158 c_prime[i] = upper[i] / m;
159 d_prime[i] = (b[i] - lower[i - 1] * d_prime[i - 1]) / m;
160 }
161
162 //backsubstitution
163 result[n - 1] = d_prime[n - 1];
164 for (int i = n - 2; i >= 0; --i) {
165 result[i] = d_prime[i] - c_prime[i] * result[i + 1];
166 }
167 }

190 std::vector<std::vector<double>> solve(double dP = 0, double ds_ = 0, double dr_ = 0) {
191 double T = Tf - T0; //Time Frame
192 double dS = S_0 / Nx; //Price step
193 double dt = T / Nt; //Time step
194 bool isCall = (type == "Call");
195 bool isAmerican = (option == "American");
196 std::vector<double> R = defineRate(dr_);
197 double Sigma = sigma + ds_;
198
199 //price grid
200 std::vector<double> S(Nx + 1);
201 for (int i = 0; i <= Nx; i++) {
202 S[i] = i * dS + dP;
203 S[i] *= std::exp(R[0] * T0); //Assuming rate is constant from today to computation date
204 }
205
206
207 std::vector<std::vector<double>> V(Nx + 1, std::vector<double>(Nt + 1, 0.0));
208
209 //maturity payoff
210 for (int i = 0; i <= Nx; i++) {
211 V[i][Nt] = isCall ? std::max(S[i] - K, 0.0) : std::max(K - S[i], 0.0);
212 V[i][Nt] *= std::exp(-trapezoidal(R, T0, Tf));
213 };
214 //Boundary conditions- Dirichlet
215 std::vector<double> subvec; // je comprends pas pourquoi vous aimez autant cette langue
216 for (int t = 0; t < Nt; t++) {
217 subvec.clear();
218 std::copy(R.begin(), R.begin() + t, std::back_inserter(subvec));
219 V[0][t] = isCall ? 0.0 : std::max(K - S[0], 0.0) * std::exp(-trapezoidal(subvec, 0, t * dt));
220 V[Nx][t] = isCall ? std::max(S[Nx] - K, 0.0) * std::exp(-trapezoidal(subvec, 0, t * dt)) : 0.0;
221 }
222
223 //Coefficient for the CN scheme
224 std::vector<double> alpha(Nx + 1), beta(Nx + 1), gamma(Nx + 1);
225
226
227 //Tridiagonal matrix component
228 std::vector<double> lower(Nx - 1), main(Nx - 1), upper(Nx - 1);
229
230
231 //Time step
232 for (int t = Nt - 1; t >= 0; --t) {
233 for (int i = 0; i <= Nx; ++i) {
234 alpha[i] = 0.25 * dt * (Sigma * Sigma * (S[i] * S[i])/ dS*dS - R[t] * S[i]/dS);
235 beta[i] = -0.5 * dt * (Sigma * Sigma * (S[i] * S[i]) / dS * dS + R[t]);
236 gamma[i] = 0.25 * dt * (Sigma * Sigma * (S[i] * S[i]) / dS * dS + R[t] * S[i]/dS);
237 }
238 for (int i = 0; i < Nx - 1; ++i) {
239 lower[i] = -alpha[i + 2];
240 main[i] = 1.0 - beta[i + 1];
241 upper[i] = -gamma[i + 1];
242 }
243
244 //Boundary adjustments
245 std::vector<double> b(Nx - 1, 0.0);
246 b[0] = alpha[1] * (V[0][t] + V[0][t + 1]) - alpha[0] * V[0][t + 1];
247 b[Nx - 2] = gamma[Nx - 1] * (V[Nx][t] + V[Nx][t + 1]);
248
249 for (int i = 1; i < Nx; ++i) {
250 b[i - 1] += (1.0 + beta[i]) * V[i][t + 1];
251 }
252
253 //Solution
254 std::vector<double> solution;
255 solveTridiagonal(lower, main, upper, b, solution);
256
257 //Update and account for American Options
258 subvec.clear();
259 std::copy(R.begin(), R.begin() + t, std::back_inserter(subvec));
260 for (int i = 1; i < Nx; i++) {
261 V[i][t] = solution[i - 1];
262
263
264 if (isAmerican) {
265 V[i][t] = std::max(V[i][t], isCall ? std::max(S[i] - K, 0.0) * std::exp(-trapezoidal(subvec, 0, t * dt)) : std::max(K - S[i], 0.0) * std::exp(-trapezoidal(subvec, 0, t * dt)));
266
267 }
268 }
269 }
270 return V;

425 double trapezoidal(const std::vector<double>& f, double a, double b) {
426 int n = f.size() - 1;
427 if (n <= 0) {
428 return 0.0;
429
430 }
431 double h = (b - a) / n; //size
432 double integral = 0.0;
433 for (int i = 0; i < n; i++) {
434 integral += (f[i] + f[i + 1]) * h / 2.0;
